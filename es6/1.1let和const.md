##let语法
```html
<script>
        // 声明变量
        let a;
        let b,c
        let e=100;
        let f =532,g='sld',l=[]

        // 1. 变量不能重复声明
        let star = 'kary'
        // 报错 防止变量污染 但是在var中没有问题
        // let star = 'lk',

        // 2. 块级作用域 全局 ，函数，eval
        {
            // if else while for 也算是块级作用域
            let girl='nihao'
        }
        // undefined
        console.log(girl);

        // 3.不存在变量提升
        // 报错
        console.log(song); 
        let song = 'caihong'

        // 4. 不影响作用域链
        {
            let school = 'shenzhen'
            function fn(){
                console.log(school);
            }
            // 调用fn函数 进入语句找school 但是没有 就向上一级去寻找school
            fn()//shenzhen

        }
    </script>
```

##let 小案例
```html
<script>
        // 获取div元素对象
        let items = document.getElementsByClassName('item')

        // 遍历并绑定事件
        for (var i = 0; i < items.length; i++) {
            items[i].onclick=function(){
                // 修改当前元素的背景颜色
                // this.style.background='pink'
                // 报错 cannot read property style
                // 因为全局变量i出问题 并没有块级作用域 所以i一直保存在全局作用域
                // 调用这个回调函数 在块里面找i 找不到就去全局 发现i=3 
                // 把var改成let 就可以
                items[i].style.background = 'pink'
            }
            // 使用let 等于在每一次执行i++时都生成了一个i的块 生成了三个i 互不影响 但是var就只生成了一个i
            // {
            //     let i=0
            // }
            // {
            //     let i=1

            // }
            // {
            //     let i=2
            // }
            
        }
    </script>
```

##const语法
```html
// 声明常量
        // 注意事项
        // 1.给初始值  
        // 2.一般常量大写
        // 3.常量值不能修改
       
        const school ='nih' 
        // 4.块级作用域
        {
            const a = 'bjhd'
        }
        console.log(a);//报错not defined
        // 5.对于数组和对象的元素修改，不算对常量的修改，不会报错
        const tea =['dad','fsfs']
        tea.push('bdauigd')//不会报错 因为常量指向的地址没有改变
        tea = 102//报错
```
