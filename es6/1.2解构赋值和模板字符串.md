##变量结构赋值
1. es6 允许按照一定模式从数组和对象中提取值，对变量进行赋值 称为结构赋值
   ```html
   <script>
        // 1.数组的解构
       const f4 = ['niko','kary','zeng']
    //    正常我们调用就是f4.n f4.k
       let [n,k,z] = f4 
       console.log(n);//niko
       console.log(k);//kary
       console.log(z);//zeng
        //2.对象的解构
        const zhao ={
            name:'niko',
            age:2,
            kary:function(){
                console.log('我可以');
            }
        };
        // let {
        //     name,
        //     age,
        //     kary
        // }=zhao
        //    正常我们调用就是f4.name f4.age
        // console.log(name);//niko
        // console.log(age);//2
        // console.log(kary);//函数
        // 就是把前面的对象去掉了 也可以单独解构
        let {name} =zhao
        console.log(name)//niko
    </script>
    ```

##模板字符串
1. es6 引入了新的声明字符串的方式 [``] '' ""
```html
 <script>
    //   1. 声明
    let str = `我也是一个字符串`
    console.log(str);

    // 2. 新特性
    // 内容中可以直接出现换行符
    // let str1 ='<ul><li></li>
    //     <li></li></ul>' 报错
    let str1 =`<ul><li></li>
        <li></li></ul>`
    // 变量拼接
    let loverst ='chj'
    let ou ='love'
    // 之前的用法
    let ou1 = loverst + 'love'
    // 现在
    let ou2 = `${ou}love`
    console.log(ou2);//lovelove

    </script>
```
